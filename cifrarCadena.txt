def convertir(cadena,desp):
    resul = "" 
    for car in cadena:
        caracterASCI = ord(car)
        caraterASCIcifrado = caracterASCI + int(desp)
        caracterUnicodeCifrado = chr(caraterASCIcifrado)
        resul += caracterUnicodeCifrado
    return resul

cadena = input("Introduzca una palabra: ")
desp = input("Desplazamiento letra(entero): ")
fraseCifrada = convertir(cadena,desp)
print("Frase cifrada:",fraseCifrada)
